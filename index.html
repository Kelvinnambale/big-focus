// Get the search input field and the search/clear icons const searchInput = document.querySelector('.search-input'); const searchIcon = document.querySelector('.search-icon'); const clearIcon = document.querySelector('.clear-icon'); // Initially, only show the search icon searchIcon.style.display = 'block'; clearIcon.style.display = 'none'; // Listen for input events on the search field searchInput.addEventListener('input', () => { // If the input is empty, show the search icon if (searchInput.value.trim() === '') { searchIcon.style.display = 'block'; clearIcon.style.display = 'none'; } else { // Otherwise, show the clear icon searchIcon.style.display = 'none'; clearIcon.style.display = 'block'; } }); // Add a click event listener to the clear icon clearIcon.addEventListener('click', () => { // Clear the search input and show the search icon searchInput.value = ''; searchIcon.style.display = 'block'; clearIcon.style.display = 'none'; });
